macro(define_testapp testapp src_list)
    set(${testapp}_DIR ${testapp})
    set(${testapp}_SRC "")
    foreach(src ${src_list})
        list(APPEND ${testapp}_SRC ${${testapp}_DIR}/${src})
    endforeach()
	add_executable(${testapp}_test ${${testapp}_SRC})
endmacro()
